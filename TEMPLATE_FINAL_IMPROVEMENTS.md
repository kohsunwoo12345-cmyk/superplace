#### 변수 추출 함수
```typescript
const extractVariables = (text: string) => {
  const regex = /#\{([^}]+)\}/g;
  const matches = text.match(regex);
  if (matches) {
    const vars = matches.map(m => m.replace(/#{|}/g, ''));
    setVariables([...new Set(vars)]); // 중복 제거
  }
};

useEffect(() => {
  extractVariables(content);
}, [content]);
```
